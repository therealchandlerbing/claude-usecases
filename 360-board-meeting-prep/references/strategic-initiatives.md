# Strategic Initiatives Workflow
## Specialized Reference for Board Packet Strategic Updates

This file is loaded just-in-time when generating the Strategic Initiatives Update component of the board packet.

---

## Purpose

Transform Asana project data, Gmail partnership communications, and organizational context into an executive assessment of progress against 360's four strategic priorities. This report connects tactical execution to strategic vision and identifies where board support is needed.

---

## The Four Strategic Priorities

From Board Motion #7 (adopted in prior meeting):

1. **Expand program reach** - Grow impact and client base
2. **Diversify funding sources** - Reduce dependency on single revenue streams
3. **Build strategic partnerships** - Focus on Brazil, China, and technology sectors
4. **Convert pipeline to revenue** - Improve close rates and shorten sales cycles

---

## Data Collection Logic

### Asana Strategic Projects

```python
def collect_strategic_initiatives_data():
    """Collect data on strategic initiatives from Asana"""

    # Identify strategic projects (by tag, custom field, or project name)
    strategic_keywords = [
        "SpacePlan", "GenIP", "ScienceLink", "NanoBioPlus",
        "CNEN", "Brazil", "China", "Strategic", "Partnership"
    ]

    portfolio_gid = "1211712180134240"
    projects = asana_get_items_for_portfolio(portfolio_gid)

    strategic_projects = []

    for project in projects:
        project_data = asana_get_project(project['gid'])

        # Check if strategic
        is_strategic = any(keyword.lower() in project_data['name'].lower()
                          for keyword in strategic_keywords)

        if is_strategic or project_data.get('client_tier') == 'Tier 1 Strategic':
            # Get milestones (sections in Asana)
            sections = asana_get_sections(project_gid=project['gid'])

            # Get tasks and completion status
            tasks = asana_get_tasks(project_gid=project['gid'])
            completed_tasks = [t for t in tasks if t['completed']]
            total_tasks = len(tasks)
            completion_rate = (len(completed_tasks) / total_tasks * 100) if total_tasks > 0 else 0

            # Get recent activity
            recent_tasks = [t for t in tasks if was_completed_this_quarter(t)]

            strategic_projects.append({
                'name': project_data['name'],
                'description': project_data.get('notes', ''),
                'completion_rate': completion_rate,
                'milestones': sections,
                'recent_activity': recent_tasks,
                'team': project_data.get('members', []),
                'last_updated': project_data['modified_at'],
                'contract_value': get_custom_field(project_data, 'contract_value'),
                'stage': get_custom_field(project_data, 'stage')
            })

    return strategic_projects
```

### Partnership Communication Analysis

```python
def analyze_partnership_communications():
    """Analyze Gmail communications with strategic partners"""

    partners = {
        'SpacePlan': ['spaceplan.com', 'enough.com'],
        'GenIP': ['genip.io'],
        'ScienceLink': ['sciencelink.org'],
        'NanoBioPlus': ['nanobioplus.com', 'curibio.com'],
        'CNEN': ['cnen.gov.br'],
        'China': ['china', '.cn']  # Broad search for China-related
    }

    communication_summary = {}

    for partner_name, domains in partners.items():
        total_interactions = 0
        last_contact = None
        key_topics = []

        for domain in domains:
            query = f'(from:*{domain}* OR to:*{domain}*) after:60daysago'
            messages = search_gmail_messages(query=query)
            total_interactions += len(messages)

            if messages and (not last_contact or messages[0]['internalDate'] > last_contact):
                last_contact = messages[0]['internalDate']

            # Extract subjects for topic analysis
            subjects = [m['subject'] for m in messages[:10]]  # Top 10 recent
            key_topics.extend(subjects)

        communication_summary[partner_name] = {
            'interaction_count': total_interactions,
            'last_contact': last_contact,
            'key_topics': key_topics[:5],  # Top 5 topics
            'engagement_level': 'High' if total_interactions > 10 else 'Medium' if total_interactions > 5 else 'Low'
        }

    return communication_summary
```

```python
def assess_strategic_priority_progress(portfolio_data, pipeline_data, revenue_data):
    """Assess progress against each of the 4 strategic priorities"""

    assessments = {}

    # Priority 1: Expand Program Reach
    new_clients_ytd = count_new_clients_this_year(portfolio_data)
    program_expansion = {
        'new_clients': new_clients_ytd,
        'new_geographies': count_new_geographies(portfolio_data),
        'new_sectors': count_new_sectors(portfolio_data),
        'status': 'üü¢' if new_clients_ytd >= 3 else 'üü°' if new_clients_ytd >= 2 else 'üî¥'
    }
    assessments['expand_reach'] = program_expansion

    # Priority 2: Diversify Funding
    revenue_by_source = revenue_data['revenue_by_category']
    top_source_pct = max(revenue_by_source.values()) / sum(revenue_by_source.values()) * 100

    diversification = {
        'revenue_sources': len(revenue_by_source),
        'top_source_concentration': top_source_pct,
        'licensing_revenue_pct': revenue_by_source.get('Licensing', 0) / sum(revenue_by_source.values()) * 100,
        'status': 'üü¢' if top_source_pct < 50 else 'üü°' if top_source_pct < 70 else 'üî¥'
    }
    assessments['diversify_funding'] = diversification

    # Priority 3: Build Strategic Partnerships
    strategic_partnerships = [p for p in portfolio_data if p['client_tier'] == 'Tier 1 Strategic']
    tier1_count = len(strategic_partnerships)

    # Check for Brazil and China presence
    has_brazil = any('brazil' in p['name'].lower() or 'cnen' in p['name'].lower()
                     for p in strategic_partnerships)
    has_china = any('china' in p['name'].lower() for p in strategic_partnerships)
    has_technology = any(tech_keyword in p['name'].lower()
                         for p in strategic_partnerships
                         for tech_keyword in ['space', 'nano', 'bio', 'tech'])

    partnerships = {
        'tier1_count': tier1_count,
        'brazil_presence': has_brazil,
        'china_presence': has_china,
        'technology_presence': has_technology,
        'status': 'üü¢' if (has_brazil and tier1_count >= 4) else 'üü°'
    }
    assessments['build_partnerships'] = partnerships

    # Priority 4: Convert Pipeline to Revenue
    conversion_metrics = calculate_conversion_metrics(pipeline_data)

    pipeline_conversion = {
        'win_rate': conversion_metrics['win_rate'],
        'avg_close_time_days': conversion_metrics['avg_close_time'],
        'proposal_to_signed_rate': conversion_metrics['proposal_to_signed'],
        'status': 'üü¢' if conversion_metrics['win_rate'] > 35 else 'üü°' if conversion_metrics['win_rate'] > 25 else 'üî¥'
    }
    assessments['convert_pipeline'] = pipeline_conversion

    return assessments
```
